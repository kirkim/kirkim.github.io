<h2>2️⃣ 함수 구현(리눅스기준)</h2>
<h4 align="middle">&#60; isalpha &#62;</h4>
```c
int isalpha(int c)
{
	if ((c >= 'A' && c <= 'Z') \
		|| (c >= 'a' && c <= 'z'))
		return (1);
	return (0);
}
```
<h4 align="middle">&#60; isalnum &#62;</h4>
```c
int isalnum(int c)
{
	if ((c >= 'A' && c <= 'Z') \
		|| (c >= 'a' && c <= 'z') \
		|| (c >= '0' && c <= '9'))
		return (1);
	return (0);
}
```
<h4 align="middle">&#60; isascii &#62;</h4>
```c
int	isascii(int c)
{
	if (c >= 0 && c <= 127)
		return (1);
	return (0);
}
```
<h4 align="middle">&#60; isdigit &#62;</h4>
```c
int	isdigit(int c)
{
	if (c >= '0' && c <= '9')
		return (1);
	return (0);
}
```
<h4 align="middle">&#60; isprint &#62;</h4>
```c
int isprint(int c)
{
	if (c >= ' ' && c <= '~')
		return (1);
	return (0);
}
```
<h4 align="middle">&#60; isupper &#62;</h4>

```c
int	isupper(int c)
```

<h4 align="middle">&#60; islower &#62;</h4>
```c
int	islower(int c)
```

<h4 align="middle">&#60; toupper &#62;</h4>
```c
int toupper(int c)
{
	if (c >= 'a' && c <= 'z')
		c -= ('a' - 'A');
	return (c);
}
```
<h4 align="middle">&#60; tolower &#62;</h4>
```c
int tolower(int c)
{
	if (c >= 'A' && c <= 'Z')
		character += ('a' - 'A');
	return (c);
}
```
